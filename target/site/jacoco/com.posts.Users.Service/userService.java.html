package com.posts.Users.Service;

import com.posts.Users.Entity.userEntity;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.web.client.RestTemplateBuilder;
import org.springframework.stereotype.Service;
import org.springframework.web.client.RestTemplate;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

@Service
public class userService {

    RestTemplate restTemplate;

    @Autowired
    public userService(RestTemplateBuilder restTemplateBuilder)
    {
        this.restTemplate = restTemplateBuilder.build();
    }

    public String userData(userEntity entity) {
        try {
            userEntity[] user = restTemplate.getForObject("http://jsonplaceholder.typicode.com/posts", userEntity[].class);
            List<userEntity> list = new ArrayList<>(Arrays.asList(user));
            List<Long> l1 = new ArrayList<>();
            l1.add(list.stream().filter(i -> i.getUserId().equals(1L)).count());
            l1.add(list.stream().filter(i -> i.getUserId().equals(2L)).count());
            l1.add(list.stream().filter(i -> i.getUserId().equals(3L)).count());
            l1.add(list.stream().filter(i -> i.getUserId().equals(4L)).count());
            l1.add(list.stream().filter(i -> i.getUserId().equals(5L)).count());
            l1.add(list.stream().filter(i -> i.getUserId().equals(6L)).count());
            l1.add(list.stream().filter(i -> i.getUserId().equals(7L)).count());
            l1.add(list.stream().filter(i -> i.getUserId().equals(8L)).count());
            l1.add(list.stream().filter(i -> i.getUserId().equals(9L)).count());
            l1.add(list.stream().filter(i -> i.getUserId().equals(10L)).count());
            return "UserId 1 : " + l1.get(0) + "\n" + "userId 2 :" + l1.get(1) + "\n" + "userId 3 :" + l1.get(2) + "\n" + "userId 4 :" + l1.get(3)
                    + "\n" + "userId 5 :" + l1.get(4) + "\n" + "userId 6 :" + l1.get(5) + "\n" + "userId 7 :" + l1.get(6)
                    + "\n" + "userId 8 :" + l1.get(7) + "\n" + "userId 9 :" + l1.get(8) + "\n" + "userId 10 :" + l1.get(9);
        }
        catch (NullPointerException e)
        {
            throw new NullPointerException("Got a Null Pointer Exception");
        }
    }


    public List<userEntity> changeData(userEntity entity) {
        try
        {
            userEntity[] user = restTemplate.getForObject("http://jsonplaceholder.typicode.com/posts", userEntity[].class);
            List<userEntity> list = new ArrayList<>(Arrays.asList((user)));
            userEntity user10 = new userEntity();
            user10.setUserId(1L);
            user10.setId(10L);
            user10.setTitle("1800Flowers");
            user10.setBody("1800Flowers");
            list.set(10, user10);
            list.remove(9);
            return list;
        }
        catch (NullPointerException e)
        {
            throw new NullPointerException("Got a NullPointer Exception");
        }
    }

}
